    uint64 
get_nproc(){
  struct proc *p;
  uint64 cnt=0;
  for(p = proc; p < &proc[NPROC]; p++) {
    acquire(&p->lock);
    if(p->state != UNUSED) {
      cnt++;
    } 
    release(&p->lock);
  }
  return cnt;
}

uint64 
get_freemem(){
  struct run *r;
  uint64 cnt=0;
  acquire(&kmem.lock);
  r=kmem.freelist;
  while(r){
    r=r->next;
    cnt++;
  }
  release(&kmem.lock);
  return cnt*PGSIZE;
}
